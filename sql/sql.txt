CREATE TABLE `tb_app` (
  `app_id` bigint(11) NOT NULL AUTO_INCREMENT,
  `app_name` varchar(255) NOT NULL,
  `app_info` varchar(255) NOT NULL COMMENT 'app的ip地址',
  `app_phone` varchar(255) NOT NULL COMMENT '开发人员电话',
  `app_state` tinyint(4) NOT NULL DEFAULT '0' COMMENT '0:未审批;1 同意;2拒绝',
  `create_time` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `modify_time` timestamp NOT NULL DEFAULT '0000-00-00 00:00:00',
  `owner_id` bigint(20) NOT NULL COMMENT 'app的开发人员',
  PRIMARY KEY (`app_id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;



CREATE TABLE `tb_group` (
  `group_id` bigint(20) NOT NULL AUTO_INCREMENT,
  `group_url` varchar(255) NOT NULL COMMENT 'Group的超链接;也就是页面的连接',
  `app_id` bigint(20) NOT NULL,
  PRIMARY KEY (`group_id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;



CREATE TABLE `tb_url` (
  `url_id` bigint(20) NOT NULL DEFAULT '0',
  `url_content` varchar(255) CHARACTER SET utf8 NOT NULL COMMENT '子页面下的超链接',
  `url_group` bigint(20) NOT NULL,
  PRIMARY KEY (`url_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;


CREATE TABLE `tb_user` (
  `user_id` bigint(11) NOT NULL AUTO_INCREMENT,
  `user_name` varchar(255) NOT NULL,
  `user_password` varchar(255) NOT NULL,
  `user_state` tinyint(4) NOT NULL DEFAULT '1' COMMENT '1:用户; 2:管理者; 3:开发者; 4:sso的超管;(当用户申请成功之后,会变成开发者)',
  `app_id` bigint(20) NOT NULL DEFAULT '0' COMMENT '默认是0,代表不属于任务app',
  PRIMARY KEY (`user_id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;


CREATE TABLE `user_group` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT,
  `user_id` bigint(20) NOT NULL,
  `group_id` bigint(20) NOT NULL,
  `app_id` bigint(20) NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;
